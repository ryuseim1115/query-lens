SELECT sub.customer_name, COUNT(p.id) AS purchase_count
FROM (
  SELECT c.id, c.name AS customer_name
  FROM   customers c
  JOIN   orders o ON c.id = o.customer_id
  WHERE  o.status = 'active'
) sub
JOIN   purchases p ON sub.id = p.customer_id
GROUP BY sub.customer_name
ORDER BY purchase_count DESC;
